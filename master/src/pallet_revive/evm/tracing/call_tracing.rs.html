17use crate::{
18	evm::{decode_revert_reason, CallLog, CallTrace, CallTracerConfig, CallType},
19	primitives::ExecReturnValue,
20	tracing::Tracing,
21	Code, DispatchError, Weight,
22};
23use alloc::{format, string::ToString, vec::Vec};
24use sp_core::{H160, H256, U256};
25
26/// A Tracer that reports logs and nested call traces transactions.
27#[derive(Default, Debug, Clone, PartialEq)]
28pub struct CallTracer<Gas, GasMapper>
29where
30	Gas: core::fmt::Debug,
31{
32	/// Map Weight to Gas equivalent.
33	gas_mapper: GasMapper,
34	/// Store all in-progress CallTrace instances.
35	traces: Vec<CallTrace<Gas>>,
36	/// Stack of indices to the current active traces.
37	current_stack: Vec<usize>,
38	/// The code and salt used to instantiate the next contract.
39	code_with_salt: Option<(Code, bool)>,
40	/// The tracer configuration.
41	config: CallTracerConfig,
42}
43
44impl<Gas: core::fmt::Debug, GasMapper> CallTracer<Gas, GasMapper> {
45	/// Create a new [`CallTracer`] instance.
46	pub fn new(config: CallTracerConfig, gas_mapper: GasMapper) -> Self {
47		Self {
48			gas_mapper,
49			traces: Vec::new(),
50			code_with_salt: None,
51			current_stack: Vec::new(),
52			config,
53		}
54	}
55
56	/// Collect the traces and return them.
57	pub fn collect_trace(mut self) -> Option<CallTrace<Gas>> {
58		self.traces.pop()
59	}
60}
61
62impl<Gas: Default + core::fmt::Debug, GasMapper: Fn(Weight) -> Gas> Tracing
63	for CallTracer<Gas, GasMapper>
64{
65	fn instantiate_code(&mut self, code: &Code, salt: Option<&[u8; 32]>) {
66		self.code_with_salt = Some((code.clone(), salt.is_some()));
67	}
68
69	fn terminate(
70		&mut self,
71		contract_address: H160,
72		beneficiary_address: H160,
73		gas_left: Weight,
74		value: U256,
75	) {
76		self.traces.last_mut().unwrap().calls.push(CallTrace {
77			from: contract_address,
78			to: beneficiary_address,
79			call_type: CallType::Selfdestruct,
80			gas: (self.gas_mapper)(gas_left),
81			value: Some(value),
82			..Default::default()
83		});
84	}
85
86	fn enter_child_span(
87		&mut self,
88		from: H160,
89		to: H160,
90		is_delegate_call: bool,
91		is_read_only: bool,
92		value: U256,
93		input: &[u8],
94		gas_left: Weight,
95	) {
96		// Increment parent's child call count.
97		if let Some(&index) = self.current_stack.last() {
98			if let Some(trace) = self.traces.get_mut(index) {
99				trace.child_call_count += 1;
100			}
101		}
102
103		if self.traces.is_empty() || !self.config.only_top_call {
104			let (call_type, input) = match self.code_with_salt.take() {
105				Some((Code::Upload(code), salt)) => (
106					if salt { CallType::Create2 } else { CallType::Create },
107					code.into_iter().chain(input.to_vec().into_iter()).collect::<Vec<_>>(),
108				),
109				Some((Code::Existing(code_hash), salt)) => (
110					if salt { CallType::Create2 } else { CallType::Create },
111					code_hash
112						.to_fixed_bytes()
113						.into_iter()
114						.chain(input.to_vec().into_iter())
115						.collect::<Vec<_>>(),
116				),
117				None => {
118					let call_type = if is_read_only {
119						CallType::StaticCall
120					} else if is_delegate_call {
121						CallType::DelegateCall
122					} else {
123						CallType::Call
124					};
125					(call_type, input.to_vec())
126				},
127			};
128
129			self.traces.push(CallTrace {
130				from,
131				to,
132				value: if is_read_only { None } else { Some(value) },
133				call_type,
134				input: input.into(),
135				gas: (self.gas_mapper)(gas_left),
136				..Default::default()
137			});
138
139			// Push the index onto the stack of the current active trace
140			self.current_stack.push(self.traces.len() - 1);
141
142		// We only track the top call, we just push a dummy index
143		} else {
144			self.current_stack.push(2);
145		}
146	}
147
148	fn log_event(&mut self, address: H160, topics: &[H256], data: &[u8]) {
149		if !self.config.with_logs {
150			return;
151		}
152
153		let current_index = self.current_stack.last().unwrap();
154
155		if let Some(trace) = self.traces.get_mut(*current_index) {
156			let log = CallLog {
157				address,
158				topics: topics.to_vec(),
159				data: data.to_vec().into(),
160				position: trace.child_call_count,
161			};
162
163			trace.logs.push(log);
164		}
165	}
166
167	fn exit_child_span(&mut self, output: &ExecReturnValue, gas_used: Weight) {
168		self.code_with_salt = None;
169
170		// Set the output of the current trace
171		let current_index = self.current_stack.pop().unwrap();
172
173		if let Some(trace) = self.traces.get_mut(current_index) {
174			trace.output = output.data.clone().into();
175			trace.gas_used = (self.gas_mapper)(gas_used);
176
177			if output.did_revert() {
178				trace.revert_reason = decode_revert_reason(&output.data);
179				trace.error = Some("execution reverted".to_string());
180			}
181
182			if self.config.only_top_call {
183				return
184			}
185
186			//  Move the current trace into its parent
187			if let Some(parent_index) = self.current_stack.last() {
188				let child_trace = self.traces.remove(current_index);
189				self.traces[*parent_index].calls.push(child_trace);
190			}
191		}
192	}
193	fn exit_child_span_with_error(&mut self, error: DispatchError, gas_used: Weight) {
194		self.code_with_salt = None;
195
196		// Set the output of the current trace
197		let current_index = self.current_stack.pop().unwrap();
198
199		if let Some(trace) = self.traces.get_mut(current_index) {
200			trace.gas_used = (self.gas_mapper)(gas_used);
201
202			trace.error = match error {
203				DispatchError::Module(sp_runtime::ModuleError { message, .. }) =>
204					Some(message.unwrap_or_default().to_string()),
205				_ => Some(format!("{:?}", error)),
206			};
207
208			if self.config.only_top_call {
209				return
210			}
211
212			//  Move the current trace into its parent
213			if let Some(parent_index) = self.current_stack.last() {
214				let child_trace = self.traces.remove(current_index);
215				self.traces[*parent_index].calls.push(child_trace);
216			}
217		}
218	}
219}
